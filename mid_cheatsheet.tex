    \section*{1. NLP Basics \& Challenges}
    \textbf{Goal:} Enable computers to understand/process human language.

    \subsection*{Types of Ambiguity}
    \begin{itemize}
        \item \textbf{Word Sense:} one word, many meanings. ``I saw her \emph{duck}'' (animal vs. verb).
        \item \textbf{PP Attachment:} which part modifier attaches to. ``I saw her duck \emph{with a telescope}'' (telescope with me or her?).
        \item \textbf{Headline/Syntactic:} compressed phrasing. ``San Jose cops kill man with knife'' (who has knife?).
        \item \textbf{Pronoun/Coref:} unclear pronoun referent. ``The umbrella can't fit in the bin because \emph{it} is too big/small'' (it = umbrella or bin?).
    \end{itemize}

    \subsection*{Modern Challenges}
    \begin{itemize}
        \item \textbf{Scalability:} 1M words (Penn Treebank) $\to$ billions (Web). Need massive corpora.
        \item \textbf{Dynamic Language:} cyber lingo, slang evolve (LOL, G2G).
        \item \textbf{Hallucination:} LLMs fabricate facts, fail basic reasoning (e.g., count 20 baseballs as 24).
    \end{itemize}

    \section*{2. Coreference \& Evaluation}
    \textbf{Coref Resolution:} group mentions (pronouns, common nouns, NEs) referring to same entity. Not all nouns/pronouns are mentions (``\emph{It} is raining'' $\to$ no referent).

    \subsection*{MUC (Link-based)}
    Computes P/R on mention \textbf{pairs/links}.
    \[P=\tfrac{TP\ pairs}{\text{predicted pairs}},\ R=\tfrac{TP\ pairs}{\text{gold pairs}}\]
    \textbf{Flaw:} putting ALL mentions in 1 cluster $\Rightarrow R=1$ artificially.

    \subsection*{B-CUBED (Mention-based)}
    Computes P/R per \emph{individual mention}, then averages. Fixes MUC giant-cluster flaw.

    \subsection*{F1 Aggregation}
    \[F1=\frac{2PR}{P+R}\]
    \begin{itemize}
        \item \textbf{Macro F1:} compute F1 per doc, then average.
        \item \textbf{Micro F1:} pool all TP/FP/FN across docs first, then single global F1.
    \end{itemize}

    \section*{3. Word Representations \& PMI}

    \subsection*{One-Hot Vectors}
    Dim $=|V|$, sparse, orthogonal. $v_{happy}\cdot v_{sad}=0$ $\Rightarrow$ no similarity notion.

    \subsection*{Taxonomy (WordNet)}
    Manual hierarchy of synonyms. Flaws: labor-intensive, misses nuance, can't track new words.

    \subsection*{Distributional Hypothesis}
    ``A word is characterized by the company it keeps.'' Build co-occurrence matrix $M$; compress via PCA/SVD (low-rank).

    \subsection*{PMI}
    \[PMI(x,y)=\log_2\frac{P(x,y)}{P(x)P(y)}\]
    Positive $\Rightarrow$ co-occur more than independent; Negative $\Rightarrow$ less.
    \textbf{PPMI:} $\max(PMI,0)$ (drop negatives, often noisy).

    \textbf{Estimate from counts} (window/doc):
    \[P(x)=\tfrac{\#x}{N},\ P(x,y)=\tfrac{\#(x,y)}{N}\]

    \section*{4. Word Embeddings (Word2Vec)}

    \subsection*{Skip-Gram}
    Given center word, predict context (outside) words in window.
    \[P(o|c)=\frac{\exp(u_o^\top v_c)}{\sum_{w\in V}\exp(u_w^\top v_c)}\]
    Loss: $J=-\sum_t\sum_{j\neq 0}\log P(w_{t+j}|w_t)$.

    \subsection*{Parameter Count}
    Each word has 2 vectors: center $v_w$, context $u_w$. Total parameters $=2|V|d$.

    \subsection*{Gradient / Update}
    \[\frac{\partial J}{\partial v_c}=u_o-\sum_w P(w|c)u_w\]
    ``Observed context minus expected context.'' Update subtracts expected weighted sum from true $u_o$.

    \subsection*{Properties}
    \begin{itemize}
        \item \textbf{Analogies} via vector arithmetic: $v_{king}-v_{man}+v_{woman}\approx v_{queen}$.
        \item \textbf{Similarity} via cosine: $\cos(a,b)=\frac{a\cdot b}{\|a\|\|b\|}$.
        \item \textbf{Biases:} embeddings encode societal stereotypes (surgeon$\to$he, nurse$\to$she).
    \end{itemize}

    \section*{5. Language Models (N-grams)}

    \subsection*{Chain Rule}
    \[P(w_1...w_T)=\prod_{t=1}^{T}P(w_t|w_1...w_{t-1})\]

    \subsection*{N-Gram MLE}
    Bigram: $P(w_i|w_{i-1})=\tfrac{C(w_{i-1},w_i)}{C(w_{i-1})}$. Trigram: condition on 2 prev.

    \subsection*{Sparsity \& Zipf}
    Zipf: $r$-th most common word has freq $\propto 1/r$. Unseen n-gram $\Rightarrow$ P=0 kills whole sequence.

    \subsection*{Smoothing}
    \textbf{Add-One (Laplace):}
    \[P=\frac{C(w_{i-1}w_i)+1}{C(w_{i-1})+V}\]
    Must add $V$ (vocab size) in denominator so probs sum to 1.

    \textbf{Add-$\lambda$:}
    \[P=\frac{C(w_{i-1}w_i)+\lambda}{C(w_{i-1})+\lambda V}\]
    $\lambda\in(0,1)$ tuned on validation; avoids stealing too much mass from seen n-grams.

    \subsection*{Evaluation}
    \textbf{Perplexity:} $PP=P(w_1..w_T)^{-1/T}=\exp(-\tfrac{1}{T}\sum\log P(w_t|\cdot))$. Lower $=$ better.

    \section*{6. Neural Language Models}

    \subsection*{Feedforward LM (Fixed Window)}
    Concatenate last $n-1$ word embeddings; feed to MLP + softmax.
    \begin{itemize}
        \item[+] Fixes sparsity (dense reps, generalizes).
        \item[--] Fixed context size; can't handle variable lengths; no param sharing across positions.
    \end{itemize}

    \subsection*{RNN LM}
    Same $W_h,W_e$ applied at \emph{every} step.
    \[h_t=\sigma(W_h h_{t-1}+W_e x_t),\ \hat y_t=\text{softmax}(Uh_t)\]
    \begin{itemize}
        \item[+] Handles variable lengths; parameter sharing; context grows.
        \item[--] Sequential (slow); \textbf{vanishing gradient} $\Rightarrow$ forgets far-away words.
    \end{itemize}

    \subsection*{LSTM}
    Adds cell state $c_t$ with gates:
    \begin{itemize}
        \item \textbf{Forget gate} $f_t$: what to drop from $c_{t-1}$.
        \item \textbf{Input gate} $i_t$: what new info to add.
        \item \textbf{Output gate} $o_t$: what to expose as $h_t$.
    \end{itemize}
    \[c_t=f_t\odot c_{t-1}+i_t\odot\tilde c_t,\ h_t=o_t\odot\tanh(c_t)\]
    Gates explicitly control memory $\Rightarrow$ mitigates vanishing gradient.

    \section*{7. Seq2Seq \& Decoding}

    \subsection*{Seq2Seq}
    Encoder RNN reads source $\to$ final hidden $=$ single fixed vector. Decoder RNN generates target conditioned on it.
    \textbf{Bottleneck:} entire source squeezed into one vector $\Rightarrow$ info loss on long sentences.

    \subsection*{Decoding Algorithms}
    \textbf{Greedy:} at each step pick $\arg\max P(w_t|\cdot)$. Fast; locally optimal, globally suboptimal.

    \textbf{Beam Search:} keep top-$k$ partial hypotheses (beam width $k$) by cumulative log-prob.
    \begin{itemize}
        \item \textbf{If $k=1$, beam search $\equiv$ greedy.}
        \item Larger $k$: better search, more compute; too large can hurt fluency.
    \end{itemize}

    \section*{8. Attention}
    \textbf{Motivation:} break Seq2Seq bottleneck by letting decoder look at \emph{all} source hidden states.

    \subsection*{Mechanism (each decoder step $t$)}
    1. Score: $e_{t,i}=s_t^\top h_i$ (dot) for each source state $h_i$.

    2. Softmax: $\alpha_{t,i}=\tfrac{\exp(e_{t,i})}{\sum_j\exp(e_{t,j})}$.

    $\Rightarrow$ \textbf{weights $\alpha\in[0,1]$, $\sum_i\alpha_{t,i}=1$.}

    3. Context: $c_t=\sum_i\alpha_{t,i}h_i$.

    4. Combine with $s_t$, predict $w_t$.

    \subsection*{Benefits}
    \begin{itemize}
        \item Solves bottleneck (direct access to all source states).
        \item Helps with vanishing gradients (shortcut paths).
        \item \textbf{Interpretable} $\alpha$ learns soft word-alignments.
    \end{itemize}

    \section*{9. Self-Attention \& Transformers}
    Removes RNN recurrence $\Rightarrow$ \textbf{parallel} processing over sequence.

    \subsection*{Q, K, V}
    Inputs $X\in\mathbb{R}^{L\times D}$. Project:
    \[Q=XW^Q,\ K=XW^K,\ V=XW^V\]
    Each token \emph{queries} others (via $Q$ vs.~$K$) and pulls info from their \emph{values} ($V$).

    \subsection*{Scaled Dot-Product}
    \[\text{Attn}(Q,K,V)=\text{softmax}\!\left(\frac{QK^\top}{\sqrt{d_k}}\right)V\]

    \subsection*{Space / Compute}
    $QK^\top$: $(L\times D)(D\times L)=L\times L$ attention matrix.
    \[\boxed{\text{Space/Time} = O(L^2)}\] Scales \textbf{quadratically} in sequence length.

    \subsection*{Extras}
    \begin{itemize}
        \item \textbf{Multi-head:} $h$ parallel $Q,K,V$ heads $\to$ concat $\to$ projection.
        \item \textbf{Positional encodings:} needed since self-attention is order-agnostic.
        \item \textbf{Masking:} decoder masks future positions to preserve autoregressive property.
    \end{itemize}

    \section*{10. Implementation Practice}

    \subsection*{HuggingFace Tokenizers}
    \begin{itemize}
        \item \texttt{from transformers import AutoTokenizer, AutoModel}
        \item \texttt{tok = AutoTokenizer.from\_pretrained("bert-base-uncased")}
        \item \texttt{ids = tok("Hello world", return\_tensors="pt")}
        \item \texttt{model = AutoModel.from\_pretrained(...)}
        \item Subword tokenization (BPE / WordPiece) handles OOV.
        \item Special tokens: \texttt{[CLS]}, \texttt{[SEP]}, \texttt{[PAD]}, \texttt{[MASK]}.
    \end{itemize}

    \section*{Quick Formulas Cheat}
    \begin{itemize}
        \item $PMI=\log_2\tfrac{P(x,y)}{P(x)P(y)}$; PPMI $=\max(0,PMI)$.
        \item W2V params: $2|V|d$. Cosine: $\tfrac{a\cdot b}{\|a\|\|b\|}$.
        \item Laplace: $\tfrac{C+1}{C(\cdot)+V}$. Add-$\lambda$: $\tfrac{C+\lambda}{C(\cdot)+\lambda V}$.
        \item Perplexity $=2^{-\frac{1}{T}\sum\log_2 P}$.
        \item F1 $=\tfrac{2PR}{P+R}$.
        \item Attention softmax sums to 1; self-attn $=O(L^2)$.
        \item Beam $k{=}1$ $\Leftrightarrow$ greedy.
    \end{itemize}

    \section*{Common Pitfall Checks}
    \begin{itemize}
        \item MUC recall $=1$ if one giant cluster $\Rightarrow$ use B-CUBED.
        \item Laplace: denom must include $+V$, not $+1$.
        \item W2V has TWO vectors per word, not one.
        \item Attention weights: after softmax, in $[0,1]$, sum to 1 (NOT raw scores).
        \item Self-attention is permutation-invariant without positional encodings.
        \item One-hot dot product $=0$ for different words $\Rightarrow$ no similarity.
    \end{itemize}
